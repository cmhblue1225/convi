// 결제 정보 암호화 const encryptPaymentData = (paymentInfo: PaymentInfo): EncryptedPaymentData => { const key = process.env.PAYMENT_ENCRYPTION_KEY; const iv = crypto.randomBytes(16); const cipher = crypto.createCipher('aes-256-cbc', key, iv); let encrypted = cipher.update(JSON.stringify({ cardNumber: paymentInfo.cardNumber, expiryDate: paymentInfo.expiryDate, cvv: paymentInfo.cvv, holderName: paymentInfo.holderName }), 'utf8', 'hex'); encrypted += cipher.final('hex'); return { encryptedData: encrypted, iv: iv.toString('hex'), timestamp: Date.now() }; }; // 결제 정보 복호화 (PCI DSS 준수) const decryptPaymentData = (encryptedData: EncryptedPaymentData): PaymentInfo => { const key = process.env.PAYMENT_ENCRYPTION_KEY; const iv = Buffer.from(encryptedData.iv, 'hex'); const decipher = crypto.createDecipher('aes-256-cbc', key, iv); let decrypted = decipher.update(encryptedData.encryptedData, 'hex', 'utf8'); decrypted += decipher.final('utf8'); return JSON.parse(decrypted); }; // 민감 정보 마스킹 const maskSensitiveData = (paymentInfo: PaymentInfo) => { return { ...paymentInfo, cardNumber: paymentInfo.cardNumber.replace(/\d(?=\d{4})/g, '*'), cvv: '***', accountNumber: paymentInfo.accountNumber?.replace(/\d(?=\d{4})/g, '*') }; };

암호화 알고리즘: AES-256-CBC

키 관리: AWS KMS 또는 HashiCorp Vault

데이터 보관: 암호화된 상태로만 저장

// 결제 무결성 검증 class PaymentValidator { static validatePaymentSignature(paymentData: PaymentData, signature: string): boolean { const expectedSignature = this.generateSignature(paymentData); return crypto.timingSafeEqual( Buffer.from(signature, 'hex'), Buffer.from(expectedSignature, 'hex') ); } static generateSignature(paymentData: PaymentData): string { const payload = [ paymentData.orderId, paymentData.amount, paymentData.timestamp, paymentData.customerKey ].join('|'); return crypto .createHmac('sha256', process.env.PAYMENT_SECRET_KEY) .update(payload) .digest('hex'); } static validateAmount(orderAmount: number, paymentAmount: number): boolean { // 부동소수점 오차 고려한 금액 비교 return Math.abs(orderAmount - paymentAmount) < 0.01; } static validatePaymentWindow(paymentTimestamp: number): boolean { const now = Date.now(); const paymentTime = new Date(paymentTimestamp).getTime(); const timeDiff = now - paymentTime; // 결제 요청 후 30분 이내만 유효 return timeDiff > 0 && timeDiff < (30 * 60 * 1000); } static async validateDuplicatePayment(orderId: string, amount: number): Promise { const existingPayment = await PaymentTransaction.findOne({ where: { orderId, amount, status: 'completed' } }); return !existingPayment; // 중복 결제가 없으면 true } }

검증 항목: 서명, 금액, 시간, 중복 방지

보안 기능: Timing attack 방지

유효 기간: 결제 요청 후 30분

// 결제 실패 처리 및 복구 class PaymentFailureHandler { static async handlePaymentFailure(error: PaymentError, context: PaymentContext) { // 실패 원인 분석 const failureAnalysis = this.analyzeFailure(error); // 실패 로그 기록 await this.logPaymentFailure({ orderId: context.orderId, customerId: context.customerId, failureCode: error.code, failureReason: error.message, amount: context.amount, paymentMethod: context.paymentMethod, timestamp: new Date(), recoverable: failureAnalysis.recoverable }); // 복구 가능한 경우 자동 재시도 if (failureAnalysis.recoverable && context.retryCount < MAX_RETRY_COUNT) { await this.scheduleRetry(context, failureAnalysis.retryDelay); return { action: 'retry_scheduled' }; } // 고객에게 대안 제시 if (failureAnalysis.suggestAlternative) { const alternatives = await this.getAlternativePaymentMethods(context); await this.notifyCustomerWithAlternatives(context.customerId, alternatives); return { action: 'alternatives_suggested' }; } // 최종 실패 처리 await this.processPaymentFailure(context); return { action: 'payment_failed' }; } private static analyzeFailure(error: PaymentError): FailureAnalysis { const recoverableErrors = [ 'NETWORK_ERROR', 'TEMPORARY_SERVER_ERROR', 'CARD_PROCESSING_ERROR' ]; const cardIssueErrors = [ 'INSUFFICIENT_BALANCE', 'CARD_EXPIRED', 'INVALID_CARD' ]; return { recoverable: recoverableErrors.includes(error.code), suggestAlternative: cardIssueErrors.includes(error.code), retryDelay: this.calculateRetryDelay(error.code), severity: this.calculateSeverity(error.code) }; } }

재시도 정책: 지수 백오프 알고리즘

대안 제시: 실패 원인별 맞춤 대안

모니터링: 실시간 실패율 추적

// 토스페이먼츠 웹훅 처리 const handleTossWebhook = async (req: Request, res: Response) => { try { // 웹훅 서명 검증 const signature = req.headers['toss-signature'] as string; const isValid = verifyWebhookSignature(req.body, signature); if (!isValid) { return res.status(400).json({ error: 'Invalid webhook signature' }); } const webhookData = req.body; // 웹훅 이벤트 처리 switch (webhookData.eventType) { case 'Payment.Approved': await handlePaymentApproved(webhookData.data); break; case 'Payment.Failed': await handlePaymentFailed(webhookData.data); break; case 'Payment.Cancelled': await handlePaymentCancelled(webhookData.data); break; case 'Refund.Completed': await handleRefundCompleted(webhookData.data); break; } // 웹훅 수신 확인 응답 res.status(200).json({ received: true }); } catch (error) { logger.error('Webhook processing failed', { error, body: req.body }); res.status(500).json({ error: 'Webhook processing failed' }); } }; const verifyWebhookSignature = (payload: any, signature: string): boolean => { const expectedSignature = crypto .createHmac('sha256', WEBHOOK_SECRET_KEY) .update(JSON.stringify(payload)) .digest('base64'); return crypto.timingSafeEqual( Buffer.from(signature, 'base64'), Buffer.from(expectedSignature, 'base64') ); };

보안 검증: HMAC-SHA256 서명 확인

이벤트 처리: 비동기 큐 시스템 활용

재시도 정책: 실패 시 exponential backoff

정산 시스템

// 자동 정산 시스템 class SettlementProcessor { static async processSettlement(paymentTransaction: PaymentTransaction) { const settlement = await this.calculateSettlement(paymentTransaction); // 정산 레코드 생성 const settlementRecord = await Settlement.create({ paymentTransactionId: paymentTransaction.id, storeId: paymentTransaction.storeId, grossAmount: paymentTransaction.amount, platformFee: settlement.platformFee, paymentFee: settlement.paymentFee, netAmount: settlement.netAmount, settlementDate: this.getSettlementDate(), status: 'pending' }); // 매장 정산 계좌에 입금 예약 await this.scheduleSettlementTransfer(settlementRecord); return settlementRecord; } private static calculateSettlement(transaction: PaymentTransaction) { const grossAmount = transaction.amount; // 플랫폼 수수료 (매출의 3%) const platformFee = Math.round(grossAmount * 0.03); // 결제 수수료 (결제사 정책에 따라) const paymentFee = this.calculatePaymentFee( transaction.paymentMethod, grossAmount ); // 매장 정산 금액 const netAmount = grossAmount - platformFee - paymentFee; return { platformFee, paymentFee, netAmount }; } private static getSettlementDate(): Date { // T+1 정산 (다음 영업일) const tomorrow = new Date(); tomorrow.setDate(tomorrow.getDate() + 1); // 주말/공휴일 제외 로직 while (this.isHolidayOrWeekend(tomorrow)) { tomorrow.setDate(tomorrow.getDate() + 1); } return tomorrow; } }

정산 주기: T+1 (다음 영업일)

수수료 체계: 플랫폼 3% + 결제사 수수료

자동화: 매일 자동 정산 처리
