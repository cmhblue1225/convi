} // 할인 한도 적용 if (tierConfig.maxDiscountAmount && discountAmount > tierConfig.maxDiscountAmount) { discountAmount = tierConfig.maxDiscountAmount; } return { totalDiscount: discountAmount, details: [{ type: 'membership', tier: membershipTier, discountAmount, description: `${membershipTier} 멤버십 할인` }] }; // 할인 조합 최적화 class DiscountOptimizer { optimizeDiscountCombination(calculations: DiscountCalculation[]): DiscountCalculation[] { // 할인 규칙에 따라 최적 조합 찾기 const compatibilityMatrix = this.buildCompatibilityMatrix(calculations); const optimizedCombination = this.findOptimalCombination(calculations, compatibilityMatrix); return optimizedCombination; } private findOptimalCombination( calculations: DiscountCalculation[], compatibilityMatrix: boolean[][] ): DiscountCalculation[] { const n = calculations.length; const combinations = this.generateCompatibleCombinations(n, compatibilityMatrix); let bestCombination = []; let maxDiscount = 0; for (const combination of combinations) { const totalDiscount = combination.reduce((sum, index) => sum + calculations[index].totalDiscount, 0 ); if (totalDiscount > maxDiscount) { maxDiscount = totalDiscount; bestCombination = combination; } } return bestCombination.map(index => calculations[index]); } // 할인 스택킹 규칙 검증 validateStackingRules(discount1: DiscountCalculation, discount2: DiscountCalculation): boolean { // 동일 유형 할인은 스택 불가 if (discount1.type === discount2.type) { return false; } // 쿠폰과 프로모션 스택킹 규칙 if (discount1.type === 'coupon' && discount2.type === 'promotion') { return discount1.stackable && discount2.stackable; } // 멤버십 할인은 다른 할인과 스택 가능 if (discount1.type === 'membership' || discount2.type === 'membership') { return true; } // 시간대별 할인 스택킹 규칙 if (discount1.type === 'time_based' || discount2.type === 'time_based') { return this.checkTimeBasedStackingRules(discount1, discount2); } return true; } // 최대 할인 한도 검증 validateDiscountLimits( discounts: DiscountCalculation[], context: DiscountContext ): DiscountCalculation[] { const totalDiscountAmount = discounts.reduce((sum, d) => sum + d.totalDiscount, 0); const orderAmount = context.order.totalAmount; // 전체 할인 한도 (예: 주문 금액의 80%) const maxDiscountRate = 0.8; const maxAllowedDiscount = orderAmount * maxDiscountRate; if (totalDiscountAmount <= maxAllowedDiscount) { return discounts; } // 할인 비율에 따라 조정 const adjustmentRatio = maxAllowedDiscount / totalDiscountAmount; return discounts.map(discount => ({ ...discount, totalDiscount: discount.totalDiscount * adjustmentRatio, adjusted: true, originalAmount: discount.totalDiscount })); } }

} // 적용 가능 상품 확인 if (coupon.applicableProducts.length > 0) { const hasApplicableProduct = order.items.some(item => coupon.applicableProducts.includes(item.productId) ); if (!hasApplicableProduct) { return { isValid: false, errorMessage: '쿠폰 적용 가능한 상품이 없습니다.' }; } } // 고객 자격 확인 if (coupon.customerRestrictions) { const isEligible = this.checkCustomerEligibility(coupon.customerRestrictions, customer); if (!isEligible) { return { isValid: false, errorMessage: '쿠폰 사용 자격이 없습니다.' }; } } return { isValid: true }; } private calculateCouponDiscount(coupon: Coupon, order: Order): number { let discountAmount = 0; switch (coupon.discountType) { case 'FIXED_AMOUNT': discountAmount = coupon.discountValue; break; case 'PERCENTAGE': const applicableAmount = this.getApplicableAmount(coupon, order); discountAmount = applicableAmount * (coupon.discountValue / 100); break; case 'FREE_SHIPPING': discountAmount = order.shippingCost || 0; break; } // 최대 할인 금액 제한 if (coupon.maxDiscountAmount && discountAmount > coupon.maxDiscountAmount) { discountAmount = coupon.maxDiscountAmount; } return Math.round(discountAmount); } }

calculateTimeBasedDiscount(order: Order, currentTime: Date): DiscountCalculation { const currentTimeString = this.formatTime(currentTime); const dayOfWeek = currentTime.getDay(); // 현재 시간에 해당하는 할인 슬롯 찾기 const activeSlot = this.timeSlots.find(slot => this.isTimeInSlot(currentTimeString, slot) && this.isValidDay(dayOfWeek, slot) ); if (!activeSlot) { return { totalDiscount: 0, details: [] }; } // 시간대별 할인 적용 가능 상품 필터링 const eligibleItems = order.items.filter(item => this.isItemEligibleForTimeDiscount(item, activeSlot) ); if (eligibleItems.length === 0) { return { totalDiscount: 0, details: [] }; } const eligibleAmount = eligibleItems.reduce((sum, item) => sum + (item.unitPrice * item.quantity), 0 ); const discountAmount = eligibleAmount * (activeSlot.discount / 100); return { totalDiscount: discountAmount, details: [{ type: 'time_based', timeSlot: activeSlot.name, discountRate: activeSlot.discount, discountAmount, description: `${activeSlot.name} 시간대 할인 ${activeSlot.discount}%` }] }; } private isTimeInSlot(currentTime: string, slot: TimeSlot): boolean { const current = this.parseTime(currentTime); const start = this.parseTime(slot.start); const end = this.parseTime(slot.end); // 자정을 넘나드는 시간대 처리 (예: 22:00 - 06:00) if (start > end) { return current >= start || current <= end; } else { return current >= start && current <= end; } } // 실시간 할인 갱신 scheduleDiscountUpdates(order: Order, callback: (discount: DiscountCalculation) => void) { const updateInterval = setInterval(() => { const currentTime = new Date();
