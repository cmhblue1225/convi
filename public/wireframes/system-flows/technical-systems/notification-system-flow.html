// 물류 이벤트 this.eventBus.on('supply.approved', this.handleSupplyApproval.bind(this)); this.eventBus.on('shipment.dispatched', this.handleShipmentDispatch.bind(this)); } async handleNewOrder(event: OrderCreatedEvent) { const { order } = event.data; // 멀티채널 알림 서비스 class MultiChannelNotificationService { private channels: Map; constructor() { this.initializeChannels(); } private initializeChannels() { this.channels.set('websocket', new WebSocketChannel()); this.channels.set('push', new PushNotificationChannel()); this.channels.set('email', new EmailChannel()); this.channels.set('sms', new SMSChannel()); this.channels.set('inapp', new InAppChannel()); } async send(notification: NotificationRequest): Promise { const results: ChannelResult[] = []; // 사용자 알림 설정 확인 const userPreferences = await this.getUserPreferences(notification.recipients); for (const channelName of notification.channels) { const channel = this.channels.get(channelName); if (!channel) continue; // 채널별 사용자 설정 필터링 const eligibleRecipients = this.filterByChannelPreferences( notification.recipients, channelName, userPreferences ); if (eligibleRecipients.length === 0) continue; try { const channelResult = await channel.send({ ...notification, recipients: eligibleRecipients }); results.push({ channel: channelName, status: 'success', deliveredCount: channelResult.deliveredCount, failedCount: channelResult.failedCount, details: channelResult.details }); } catch (error) { results.push({ channel: channelName, status: 'error', error: error.message, deliveredCount: 0, failedCount: eligibleRecipients.length }); // 실패한 알림을 재시도 큐에 추가 await this.queueForRetry(notification, channelName, eligibleRecipients); } } // 전송 결과 로깅 및 분석 await this.logNotificationResults(notification, results); return { notificationId: notification.id, totalRecipients: notification.recipients.length, channelResults: results, overallStatus: this.calculateOverallStatus(results) }; } private filterByChannelPreferences( recipients: string[], channel: string, preferences: UserPreferences[] ): string[] { return recipients.filter(recipientId => { const userPref = preferences.find(p => p.userId === recipientId); if (!userPref) return true; // 기본값: 허용 return userPref.notificationSettings[channel] !== false; }); } }

// WebSocket 실시간 알림 채널 class WebSocketChannel implements NotificationChannel { private io: Server; private activeConnections: Map; constructor(server: HttpServer) { this.io = new Server(server, { cors: { origin: "*" }, transports: ['websocket', 'polling'] }); this.setupConnectionHandlers(); } private setupConnectionHandlers() { this.io.on('connection', (socket: Socket) => { socket.on('authenticate', async (token: string) => { try { const user = await this.authenticateUser(token); socket.join(`user:${user.id}`); socket.join(`role:${user.role}`); // 사용자별 활성 연결 추적 this.activeConnections.set(user.id, socket); console.log(`User ${user.id} connected via WebSocket`); // 미읽은 알림 전송 await this.sendPendingNotifications(user.id, socket); } catch (error) { socket.emit('auth_error', { message: 'Authentication failed' }); socket.disconnect(); } }); socket.on('disconnect', () => { // 연결 해제 시 정리 for (const [userId, userSocket] of this.activeConnections) { if (userSocket === socket) { this.activeConnections.delete(userId); break; } } }); socket.on('notification_read', async (notificationId: string) => { await this.markNotificationAsRead(notificationId); }); }); } async send(notification: NotificationRequest): Promise { const results = []; for (const recipientId of notification.recipients) { try { // 실시간 전송 시도 const delivered = await this.sendToUser(recipientId, notification); if (delivered) { results.push({ recipientId, status: 'delivered' }); } else { // 오프라인 사용자 - 다음 로그인 시 전송을 위해 저장 await this.storeOfflineNotification(recipientId, notification); results.push({ recipientId, status: 'stored_offline' }); } } catch (error) { results.push({ recipientId, status: 'failed', error: error.message }); } } const deliveredCount = results.filter(r => r.status === 'delivered').length; const failedCount = results.filter(r => r.status === 'failed').length; return { deliveredCount, failedCount, details: results }; } private async sendToUser(userId: string, notification: NotificationRequest): Promise { const socket = this.activeConnections.get(userId); if (!socket || !socket.connected) { return false; } // 알림 데이터 포맷팅 const formattedNotification = { id: generateUUID(), type: notification.type, subtype: notification.subtype, title: await this.generateTitle(notification), message: await this.generateMessage(notification), data: notification.data, timestamp: new Date().toISOString(), priority: notification.priority, actions: await this.generateActions(notification) }; // 실시간 전송 socket.emit('notification', formattedNotification); // 전송 확인 대기 (optional) return new Promise((resolve) => { const timeout = setTimeout(() => resolve(false), 5000); socket.once(`notification_received:${formattedNotification.id}`, () => { clearTimeout(timeout); resolve(true); }); }); } }

// 멀티플랫폼 푸시 알림 class PushNotificationChannel implements NotificationChannel { private apnsProvider: apn.Provider; private fcmAdmin: admin.messaging.Messaging; private webPushClient: webpush; constructor() { this.initializeProviders(); } private initializeProviders() { // Apple Push Notification Service this.apnsProvider = new apn.Provider({ token: { key: process.env.APNS_KEY_PATH, keyId: process.env.APNS_KEY_ID, teamId: process.env.APNS_TEAM_ID }, production: process.env.NODE_ENV === 'production' }); // Firebase Cloud Messaging this.fcmAdmin = admin.messaging(); // Web Push webpush.setVapidDetails( process.env.VAPID_SUBJECT, process.env.VAPID_PUBLIC_KEY, process.env.VAPID_PRIVATE_KEY ); } async send(notification: NotificationRequest): Promise { const results = []; // 수신자별 디바이스 토큰 조회 const deviceTokens = await this.getDeviceTokens(notification.recipients); for (const [userId, tokens] of deviceTokens) { for (const token of tokens) { try { let result; switch (token.platform) { case 'ios': result = await this.sendAPNS(token, notification); break; case 'android': result = await this.sendFCM(token, notification); break; case 'web': result = await this.sendWebPush(token, notification); break; } results.push({ userId, tokenId: token.id, platform: token.platform, status: result.success ? 'delivered' : 'failed', error: result.error }); } catch (error) { results.push({ userId, tokenId: token.id, platform: token.platform, status: 'failed', error: error.message }); // 무효한 토큰 처리 if (this.isInvalidTokenError(error)) { await this.removeInvalidToken(token.id); } } } } const deliveredCount = results.filter(r => r.status === 'delivered').length; const failedCount = results.filter(r => r.status === 'failed').length; return { deliveredCount, failedCount, details: results }; } private async sendAPNS(token: DeviceToken, notification: NotificationRequest) { const message = new apn.Notification({ alert: { title: await this.generateTitle(notification), body: await this.generateMessage(notification) }, badge: await this.calculateBadgeCount(token.userId), sound: this.getNotificationSound(notification.priority), payload: { type: notification.type, subtype: notification.subtype, data: notification.data }, contentAvailable: true, mutableContent: true, category: notification.type, threadId: notification.threadId }); const result = await this.apnsProvider.send(message, token.value); return { success: result.sent.length > 0, error: result.failed.length > 0 ? result.failed[0].error : null }; } private async sendFCM(token: DeviceToken, notification: NotificationRequest) { const message: admin.messaging.Message = { token: token.value, notification: { title: await this.generateTitle(notification), body: await this.generateMessage(notification), imageUrl: notification.imageUrl }, data: { type: notification.type, subtype: notification.subtype, payload: JSON.stringify(notification.data) }, android: { priority: notification.priority === 'high' ? 'high' : 'normal', notification: { sound: this.getNotificationSound(notification.priority), channelId: notification.type, priority: notification.priority === 'high' ? 'high' : 'default' } }, apns: { payload: { aps: { badge: await this.calculateBadgeCount(token.userId), sound: this.getNotificationSound(notification.priority) } } } }; const result = await this.fcmAdmin.send(message); return { success: !!result, messageId: result }; } }

} // 카테고리별 설정 확인 if (!preferences.categories[notification.type]) { return { send: false, reason: 'category_disabled' }; } // 조용한 시간 확인 if (this.isQuietHours(preferences.quietHours, userId)) { // 긴급 알림은 조용한 시간에도 전송 if (notification.priority !== 'urgent') { return { send: false, reason: 'quiet_hours', reschedule: this.getNextAvailableTime(preferences.quietHours) }; const now = moment().tz(quietHours.timezone); const start = moment(quietHours.startTime, 'HH:mm').tz(quietHours.timezone); const end = moment(quietHours.endTime, 'HH:mm').tz(quietHours.timezone); // 자정을 넘어가는 경우 처리 if (start.isAfter(end)) { return now.isAfter(start) || now.isBefore(end); } else { return now.isBetween(start, end); } } }
