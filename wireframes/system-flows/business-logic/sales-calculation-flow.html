// 실시간 매출 계산 시스템 class SalesCalculationEngine { async calculateOrderSales(orderId: string): Promise { const order = await Order.findById(orderId).populate('items'); let totalRevenue = 0; let totalCost = 0; let totalProfit = 0; let totalTax = 0; const itemSales = []; // 상품별 매출 계산 for (const item of order.items) { const product = await Product.findById(item.productId); const itemSale = { productId: item.productId, productName: product.name, quantity: item.quantity, unitPrice: item.unitPrice, unitCost: product.costPrice, revenue: item.quantity * item.unitPrice, cost: item.quantity * product.costPrice, profit: (item.quantity * item.unitPrice) - (item.quantity * product.costPrice), margin: ((item.unitPrice - product.costPrice) / item.unitPrice) * 100, discount: item.discountAmount || 0 }; itemSales.push(itemSale); totalRevenue += itemSale.revenue - itemSale.discount; totalCost += itemSale.cost; totalProfit += itemSale.profit - itemSale.discount; } // 주문 할인 적용 const orderDiscount = order.discountAmount || 0; totalRevenue -= orderDiscount; totalProfit -= orderDiscount; // 세금 계산 totalTax = this.calculateTax(totalRevenue, order.storeId); // 수수료 계산 const fees = await this.calculateFees(order); // 순매출 계산 const netRevenue = totalRevenue - totalTax - fees.total; return { orderId, itemSales, totals: { grossRevenue: totalRevenue + orderDiscount, discount: orderDiscount + itemSales.reduce((sum, item) => sum + item.discount, 0), netRevenue: totalRevenue, totalCost, grossProfit: totalProfit, tax: totalTax, fees: fees.total, netProfit: totalProfit - totalTax - fees.total }, timestamp: new Date() }; } private calculateTax(revenue: number, storeId: string): number { // 부가세 (10%) const vatRate = 0.10; return Math.round(revenue * vatRate); } private async calculateFees(order: Order): Promise { const platformFeeRate = 0.03; // 플랫폼 수수료 3% const paymentFeeRate = this.getPaymentFeeRate(order.paymentMethod); const platformFee = Math.round(order.totalAmount * platformFeeRate); const paymentFee = Math.round(order.totalAmount * paymentFeeRate); const deliveryFee = order.deliveryFee || 0; return { platform: platformFee, payment: paymentFee, delivery: deliveryFee, total: platformFee + paymentFee + deliveryFee }; } }

(profit / totalRevenue) * 100 : 0, averageOrderValue: orders.length > 0 ? totalRevenue / orders.length : 0 }; } private async generateRealtimeAlerts( storeId: string, today: DayMetrics, yesterday: DayMetrics ): Promise { const alerts = []; // 매출 급감 알림 const revenueDecline = yesterday.revenue > 0 ? ((today.revenue - yesterday.revenue) / yesterday.revenue) * 100 : 0; if (revenueDecline < -20) { alerts.push({ type: 'warning', title: '매출 급감', message: `어제 대비 매출이 ${Math.abs(revenueDecline).toFixed(1)}% 감소했습니다.`, severity: 'high' }); } // 목표 달성 알림 const dailyTarget = await this.getDailyTarget(storeId); const targetProgress = (today.revenue / dailyTarget) * 100; if (targetProgress >= 100) { alerts.push({ type: 'success', title: '일일 목표 달성', message: `오늘 매출 목표를 ${targetProgress.toFixed(1)}% 달성했습니다!`, severity: 'info' }); } return alerts; } }

} // 세무 신고 데이터 생성 await this.generateTaxFilingData(storeId, year, month, { vatCalculation, incomeTaxCalculation, localTaxCalculation }); return settlement; } private calculateVAT(monthlyData: MonthlySummary): TaxCalculation { const vatRate = 0.10; // 10% 부가세 const taxableAmount = monthlyData.netRevenue; const vatAmount = Math.round(taxableAmount * vatRate); return { taxType: 'VAT', rate: vatRate, taxableAmount, amount: vatAmount, dueDate: this.getVATDueDate(monthlyData.year, monthlyData.month), fillingRequired: vatAmount > 0 }; } private calculateIncomeTax(monthlyData: MonthlySummary): TaxCalculation { // 소득세는 연간 합산 후 계산하므로 월별 추정치 제공 const estimatedAnnualProfit = monthlyData.netProfit * 12; const incomeTaxRate = this.getIncomeTaxRate(estimatedAnnualProfit); const monthlyEstimate = Math.round((estimatedAnnualProfit * incomeTaxRate) / 12); return { taxType: 'IncomeTax', rate: incomeTaxRate, taxableAmount: monthlyData.netProfit, amount: monthlyEstimate, dueDate: this.getIncomeTaxDueDate(monthlyData.year), fillingRequired: true, isEstimate: true }; } private getIncomeTaxRate(annualProfit: number): number { // 소득세 누진세율 적용 if (annualProfit <= 12000000) return 0.06; // 6% if (annualProfit <= 46000000) return 0.15; // 15% if (annualProfit <= 88000000) return 0.24; // 24% if (annualProfit <= 150000000) return 0.35; // 35% if (annualProfit <= 300000000) return 0.38; // 38% if (annualProfit <= 500000000) return 0.40; // 40% return 0.42; // 42% } async generateTaxFilingData( storeId: string, year: number, month: number, taxes: TaxCalculations ): Promise { const store = await Store.findById(storeId); const monthlyData = await MonthlySalesSummary.findOne({ storeId, year, month }); return { businessInfo: { businessNumber: store.businessNumber, businessName: store.name, ownerName: store.ownerName, address: store.address }, period: { year, month }, revenue: { gross: monthlyData.grossRevenue, net: monthlyData.netRevenue, taxable: monthlyData.netRevenue }, expenses: { cost: monthlyData.totalCost, fees: monthlyData.totalFees, other: monthlyData.otherExpenses || 0 }, taxes: { vat: taxes.vatCalculation, incomeTax: taxes.incomeTaxCalculation, localTax: taxes.localTaxCalculation }, filingDeadlines: { vatDue: taxes.vatCalculation.dueDate, incomeTaxDue: taxes.incomeTaxCalculation.dueDate }, documents: await this.generateTaxDocuments(storeId, year, month) }; 📊 매출 계산 공식 및 지표

💰 기본 매출 지표

  • 총매출: 상품가격 × 수량
  • 순매출: 총매출 - 할인
  • 매출원가: 원가 × 수량
  • 매출총이익: 순매출 - 매출원가
  • 순이익: 매출총이익 - 세금 - 수수료

📈 수익성 지표

  • 매출총이익률: (매출총이익 ÷ 순매출) × 100
  • 순이익률: (순이익 ÷ 순매출) × 100
  • 평균 주문가: 총매출 ÷ 주문수
  • 객단가: 총매출 ÷ 고객수
  • 평방미터당 매출: 총매출 ÷ 매장면적

🏆 성과 지표

  • 매출 성장률: ((이번 기간 - 이전 기간) ÷ 이전 기간) × 100
  • 재고 회전율: 매출원가 ÷ 평균재고
  • 고객 재방문율: 재방문 고객 ÷ 전체 고객 × 100
  • 상품별 기여도: 상품매출 ÷ 총매출 × 100
  • 시간대별 매출비중: 시간대매출 ÷ 일매출 × 100

💸 세금 및 수수료

  • 부가세: 과세매출 × 10%
  • 플랫폼 수수료: 총매출 × 3%
  • 결제 수수료: 결제수단별 차등
  • 소득세: 소득구간별 누진세율
  • 지방세: 소득세 × 10%
