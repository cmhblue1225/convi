// 포인트 적립 계산 엔진 class PointsCalculationEngine { private rules: PointsRule[] = []; private bonusMultipliers: BonusMultiplier[] = []; async calculatePoints( transaction: Transaction, customer: Customer ): Promise { // 기본 포인트 계산 const basePoints = this.calculateBasePoints(transaction, customer); // 멤버십 등급별 포인트 계산 const tierPoints = this.calculateTierPoints(basePoints, customer.membershipTier); // 보너스 포인트 계산 const bonusPoints = await this.calculateBonusPoints(transaction, customer); // 특별 이벤트 포인트 계산 const eventPoints = await this.calculateEventPoints(transaction, customer); // 총 포인트 합산 const totalPoints = tierPoints + bonusPoints + eventPoints; // 포인트 한도 적용 const finalPoints = this.applyPointsLimits(totalPoints, customer); return { basePoints, tierPoints, bonusPoints, eventPoints, totalPoints: finalPoints, breakdown: this.generatePointsBreakdown({ basePoints, tierPoints, bonusPoints, eventPoints }), expiryDate: this.calculateExpiryDate() }; } private calculateBasePoints(transaction: Transaction, customer: Customer): number { // 비회원 기본 적립률: 0.5% // 회원 기본 적립률: 1% const baseRate = customer.isMember() ? 0.01 : 0.005; // 포인트 적립 대상 금액 (할인 후 금액) const eligibleAmount = this.getEligibleAmount(transaction); return Math.floor(eligibleAmount * baseRate); } private calculateTierPoints(basePoints: number, tier: MembershipTier): number { const tierMultipliers = { 'basic': 1.0, // 기본 적립 'silver': 1.2, // 20% 추가 'gold': 1.5, // 50% 추가 'platinum': 2.0 // 100% 추가 }; const multiplier = tierMultipliers[tier] || 1.0; return Math.floor(basePoints * multiplier); } private async calculateBonusPoints( transaction: Transaction, customer: Customer ): Promise { let bonusPoints = 0; // 카테고리별 보너스 bonusPoints += this.calculateCategoryBonus(transaction); // 상품별 보너스 bonusPoints += this.calculateProductBonus(transaction); // 결제수단별 보너스 bonusPoints += this.calculatePaymentMethodBonus(transaction); // 시간대별 보너스 bonusPoints += this.calculateTimeBasedBonus(transaction); // 구매량별 보너스 bonusPoints += this.calculateVolumeBonus(transaction); return bonusPoints; } }

// 다양한 보너스 포인트 계산 class BonusPointsCalculator { calculateCategoryBonus(transaction: Transaction): number { let bonusPoints = 0; const categoryBonusRules = this.getCategoryBonusRules(); for (const item of transaction.items) { const categoryRule = categoryBonusRules.find(rule => rule.categoryIds.includes(item.categoryId) && this.isRuleActive(rule) ); if (categoryRule) { const itemAmount = item.unitPrice * item.quantity; bonusPoints += Math.floor(itemAmount * (categoryRule.bonusRate / 100)); } } return bonusPoints; } calculateProductBonus(transaction: Transaction): number { let bonusPoints = 0; const productBonusRules = this.getProductBonusRules(); for (const item of transaction.items) { const productRule = productBonusRules.find(rule => rule.productIds.includes(item.productId) && this.isRuleActive(rule) ); if (productRule) { const itemAmount = item.unitPrice * item.quantity; switch (productRule.bonusType) { case 'FIXED_POINTS': bonusPoints += productRule.bonusValue * item.quantity; break; case 'PERCENTAGE': bonusPoints += Math.floor(itemAmount * (productRule.bonusValue / 100)); break; case 'MULTIPLIER': const basePoints = Math.floor(itemAmount * 0.01); // 기본 1% bonusPoints += Math.floor(basePoints * (productRule.bonusValue - 1)); break; } } } return bonusPoints; } calculatePaymentMethodBonus(transaction: Transaction): number { const paymentBonusRules = this.getPaymentMethodBonusRules(); const rule = paymentBonusRules.find(rule => rule.paymentMethods.includes(transaction.paymentMethod) && this.isRuleActive(rule) ); if (!rule) return 0; const transactionAmount = transaction.totalAmount; switch (rule.bonusType) { case 'FIXED_POINTS': return rule.bonusValue; case 'PERCENTAGE': return Math.floor(transactionAmount * (rule.bonusValue / 100)); default: return 0; } } calculateTimeBasedBonus(transaction: Transaction): number { const currentTime = new Date(); const timeBasedRules = this.getTimeBasedBonusRules(); const activeRule = timeBasedRules.find(rule => this.isTimeInRange(currentTime, rule.startTime, rule.endTime) && this.isDayOfWeekMatching(currentTime, rule.daysOfWeek) && this.isRuleActive(rule) ); if (!activeRule) return 0; const eligibleAmount = this.getEligibleAmountForTimeBonus(transaction, activeRule); return Math.floor(eligibleAmount * (activeRule.bonusRate / 100)); } calculateVolumeBonus(transaction: Transaction): number { const volumeBonusRules = this.getVolumeBonusRules(); let bonusPoints = 0; for (const rule of volumeBonusRules) { if (!this.isRuleActive(rule)) continue; const qualifying = this.checkVolumeQualification(transaction, rule); if (qualifying.isQualified) { switch (rule.bonusType) { case 'FIXED_POINTS_PER_TIER': bonusPoints += rule.bonusValue * qualifying.tierCount; break; case 'PERCENTAGE_ON_EXCESS': const excessAmount = qualifying.excessAmount; bonusPoints += Math.floor(excessAmount * (rule.bonusValue / 100)); break; case 'MULTIPLIER_ON_TOTAL': const totalAmount = transaction.totalAmount; const basePoints = Math.floor(totalAmount * 0.01); bonusPoints += Math.floor(basePoints * (rule.bonusValue - 1)); break; } } } return bonusPoints; } }

// 포인트 적립 한도 관리 시스템 class PointsLimitManager { applyPointsLimits(points: number, customer: Customer): number { // 1. 일회 적립 한도 확인 const singleTransactionLimit = this.getSingleTransactionLimit(customer); let adjustedPoints = Math.min(points, singleTransactionLimit); // 2. 일일 적립 한도 확인 const dailyLimit = this.getDailyLimit(customer); const todayAccrued = this.getTodayAccruedPoints(customer); const remainingDailyLimit = Math.max(0, dailyLimit - todayAccrued); adjustedPoints = Math.min(adjustedPoints, remainingDailyLimit); // 3. 월간 적립 한도 확인 const monthlyLimit = this.getMonthlyLimit(customer); const monthlyAccrued = this.getMonthlyAccruedPoints(customer); const remainingMonthlyLimit = Math.max(0, monthlyLimit - monthlyAccrued); adjustedPoints = Math.min(adjustedPoints, remainingMonthlyLimit); // 4. 연간 적립 한도 확인 (고액 거래 대응) const yearlyLimit = this.getYearlyLimit(customer); const yearlyAccrued = this.getYearlyAccruedPoints(customer); const remainingYearlyLimit = Math.max(0, yearlyLimit - yearlyAccrued); adjustedPoints = Math.min(adjustedPoints, remainingYearlyLimit); return adjustedPoints; } private getSingleTransactionLimit(customer: Customer): number { // 멤버십 등급별 일회 적립 한도 const limits = { 'guest': 1000, // 비회원: 1,000 포인트 'basic': 2000, // Basic: 2,000 포인트 'silver': 5000, // Silver: 5,000 포인트 'gold': 10000, // Gold: 10,000 포인트 'platinum': 20000 // Platinum: 20,000 포인트 }; const tier = customer.membershipTier || 'guest'; return limits[tier] || limits['guest']; } private getDailyLimit(customer: Customer): number { // 멤버십 등급별 일일 적립 한도 const limits = { 'guest': 2000, // 비회원: 2,000 포인트 'basic': 5000, // Basic: 5,000 포인트 'silver': 10000, // Silver: 10,000 포인트 'gold': 20000, // Gold: 20,000 포인트 'platinum': 50000 // Platinum: 50,000 포인트 }; const tier = customer.membershipTier || 'guest'; return limits[tier] || limits['guest']; } async checkPointsLimitWarnings(customer: Customer): Promise { const warnings = []; // 일일 한도 경고 (80% 초과 시) const dailyLimit = this.getDailyLimit(customer); const todayAccrued = await this.getTodayAccruedPoints(customer); const dailyUsageRate = todayAccrued / dailyLimit; if (dailyUsageRate > 0.8) { warnings.push({ type: 'daily_limit_warning', message: `일일 포인트 적립 한도의 ${Math.round(dailyUsageRate * 100)}%를 사용했습니다.`, remaining: dailyLimit - todayAccrued, severity: dailyUsageRate > 0.9 ? 'high' : 'medium' }); } // 월간 한도 경고 const monthlyLimit = this.getMonthlyLimit(customer); const monthlyAccrued = await this.getMonthlyAccruedPoints(customer); const monthlyUsageRate = monthlyAccrued / monthlyLimit; if (monthlyUsageRate > 0.8) { warnings.push({ type: 'monthly_limit_warning', message: `월간 포인트 적립 한도의 ${Math.round(monthlyUsageRate * 100)}%를 사용했습니다.`, remaining: monthlyLimit - monthlyAccrued, severity: monthlyUsageRate > 0.9 ? 'high' : 'medium' }); } return warnings; } }

// 포인트 사용 및 차감 시스템 class PointsUsageManager { async processPointsUsage( customerId: string, usageAmount: number, transactionId: string, usageType: string = 'purchase' ): Promise { // 사용 가능 포인트 확인 const availablePoints = await this.getAvailablePoints(customerId); if (usableAmount > 0) { // 포인트 기록에서 차감 await trx('points_records') .where('id', record.id) .decrement('balance', usableAmount); // 포인트 기록이 모두 사용된 경우 상태 변경 if (record.balance === usableAmount) { await trx('points_records') .where('id', record.id) .update({ status: 'used_up' }); } usageDetails.push({ pointsRecordId: record.id, usedAmount: usableAmount, earnedDate: record.earnedDate, expiryDate: record.expiryDate }); remainingUsage -= usableAmount; } } // 고객 총 포인트에서 차감 await trx('customers') .where('id', customerId) .decrement('total_points', usageAmount); // 사용 이력 기록 await trx('points_usage_history').insert({ customer_id: customerId, transaction_id: transactionId, usage_type: usageType, used_amount: usageAmount, usage_details: JSON.stringify(usageDetails), used_at: new Date() }); await trx.commit(); return { success: true, usedAmount: usageAmount, remainingPoints: availablePoints - usageAmount, usageDetails }; } catch (error) { await trx.rollback(); throw error; } } async calculatePointsValue(points: number): Promise { // 포인트 가치 계산 (예: 1포인트 = 1원) const pointValue = await this.getPointValue(); return points * pointValue; } async suggestOptimalPointsUsage( customerId: string, orderAmount: number ): Promise { const availablePoints = await this.getAvailablePoints(customerId); const pointValue = await this.getPointValue(); suggestedUsage = Math.min(optimalUsage, availablePoints); reason = '포인트를 적절히 활용하여 할인 혜택을 받으세요.'; } return { suggestedAmount: suggestedUsage, discountValue: suggestedUsage * pointValue, reason, availablePoints, expiringPoints: expiringAmount, maxUsableAmount }; } }
