const ttl = options?.ttl || 3600; // 모든 계층에 저장 (write-through) await Promise.all([ this.l1Cache.set(cacheKey, value, Math.min(ttl, 300)), // L1: 최대 5분 this.l2Cache.set(cacheKey, value, ttl), // L2: 설정된 TTL this.l3Cache.set(cacheKey, value, ttl * 2) // L3: 2배 TTL ]); return false; } } async ttl(key: string): Promise { try { return await this.redis.ttl(key); } catch (error) { console.error(`Redis ttl error for key ${key}:`, error); return -1; } } async keys(pattern: string): Promise { try { return await this.redis.keys(pattern); // 고성능 인메모리 캐시 class MemoryCache { private cache: Map; private maxSize: number; private cleanupInterval: NodeJS.Timeout; constructor(options: MemoryCacheOptions = {}) { this.cache = new Map(); this.maxSize = options.maxSize || 10000; this.startCleanupTimer(); } get(key: string): T | null { const entry = this.cache.get(key); if (!entry) return null; // TTL 확인 if (entry.expiresAt && Date.now() > entry.expiresAt) { this.cache.delete(key); return null; } // LRU 업데이트 entry.lastAccessed = Date.now(); return entry.value as T; } set(key: string, value: T, ttl?: number): void { // 크기 제한 확인 if (this.cache.size >= this.maxSize && !this.cache.has(key)) { this.evictLRU(); } const now = Date.now(); const entry: CacheEntry = { value, createdAt: now, lastAccessed: now, expiresAt: ttl ? now + (ttl * 1000) : undefined }; this.cache.set(key, entry); } delete(key: string): boolean { return this.cache.delete(key); } clear(): void { this.cache.clear(); } size(): number { return this.cache.size; } // LRU 기반 캐시 제거 private evictLRU(): void { let oldestKey = ''; let oldestTime = Date.now(); for (const [key, entry] of this.cache.entries()) { if (entry.lastAccessed < oldestTime) { oldestTime = entry.lastAccessed; oldestKey = key; } } if (oldestKey) { this.cache.delete(oldestKey); } } // 주기적인 만료된 캐시 정리 private startCleanupTimer(): void { this.cleanupInterval = setInterval(() => { const now = Date.now(); const keysToDelete: string[] = []; for (const [key, entry] of this.cache.entries()) { if (entry.expiresAt && now > entry.expiresAt) { keysToDelete.push(key); } } keysToDelete.forEach(key => this.cache.delete(key)); }, 60000); // 1분마다 정리 } // 캐시 통계 getStats(): MemoryCacheStats { const now = Date.now(); let expiredCount = 0; let totalAge = 0; for (const entry of this.cache.values()) { if (entry.expiresAt && now > entry.expiresAt) { expiredCount++; } totalAge += now - entry.createdAt; } return { totalEntries: this.cache.size, expiredEntries: expiredCount, averageAge: this.cache.size > 0 ? totalAge / this.cache.size : 0, memoryUsage: this.estimateMemoryUsage() }; } private estimateMemoryUsage(): number { // 간단한 메모리 사용량 추정 let usage = 0; for (const [key, entry] of this.cache.entries()) { usage += key.length * 2; // Unicode 문자당 2바이트 usage += JSON.stringify(entry.value).length * 2; usage += 64; // 메타데이터 추정 } return usage; } destroy(): void { if (this.cleanupInterval) { clearInterval(this.cleanupInterval); } this.cache.clear(); } }

this.eventBus = eventBus; this.setupEventListeners(); } // 이벤트 기반 자동 무효화 설정 private setupEventListeners(): void { // 상품 정보 변경 시 관련 캐시 무효화 this.eventBus.on('product.updated', async (event: ProductUpdatedEvent) => { await this.invalidateProductCaches(event.productId); }); // 가격 변경 시 관련 캐시 무효화 this.eventBus.on('price.updated', async (event: PriceUpdatedEvent) => { await this.invalidatePriceCaches(event.productId); }); // 재고 변경 시 관련 캐시 무효화 this.eventBus.on('inventory.updated', async (event: InventoryUpdatedEvent) => { await this.invalidateInventoryCaches(event.storeId, event.productId); }); // 사용자 정보 변경 시 관련 캐시 무효화 this.eventBus.on('user.updated', async (event: UserUpdatedEvent) => { await this.invalidateUserCaches(event.userId); }); } async invalidateProductCaches(productId: string): Promise { const patterns = [ `product:${productId}:*`, `products:category:*`, `products:search:*`, `products:featured:*` ]; await Promise.all( patterns.map(pattern => this.cacheManager.invalidate(pattern)) ); // 태그 기반 무효화 await this.cacheManager.invalidate(['products', `product_${productId}`]); } async invalidatePriceCaches(productId: string): Promise { const patterns = [ `price:${productId}:*`, `prices:*`, `cart:*`, // 장바구니에 가격 정보가 포함됨 `order:*:total` // 주문 총액 계산 캐시 ]; await Promise.all( patterns.map(pattern => this.cacheManager.invalidate(pattern)) ); } async invalidateInventoryCaches(storeId: string, productId: string): Promise { const patterns = [ `inventory:${storeId}:${productId}`, `inventory:${storeId}:*`, `stock:${storeId}:*`, `availability:${productId}:*` ]; await Promise.all( patterns.map(pattern => this.cacheManager.invalidate(pattern)) ); } async invalidateUserCaches(userId: string): Promise { const patterns = [ `user:${userId}:*`, `session:${userId}:*`, `cart:${userId}:*`, `orders:${userId}:*`, `points:${userId}:*` ]; await Promise.all( patterns.map(pattern => this.cacheManager.invalidate(pattern)) ); } // 계층적 무효화 (상위 카테고리 변경 시 하위 모든 캐시 무효화) async invalidateHierarchical(entityType: string, entityId: string): Promise { switch (entityType) { case 'category': await this.invalidateCategoryHierarchy(entityId); break; case 'store': await this.invalidateStoreHierarchy(entityId); break; case 'promotion': await this.invalidatePromotionHierarchy(entityId); break; } } private async invalidateCategoryHierarchy(categoryId: string): Promise { // 카테고리 계층 구조에 따른 캐시 무효화 const childCategories = await this.getChildCategories(categoryId); const patterns = [ `category:${categoryId}:*`, `products:category:${categoryId}:*`, ...childCategories.map(id => `category:${id}:*`), ...childCategories.map(id => `products:category:${id}:*`) ]; await Promise.all( patterns.map(pattern => this.cacheManager.invalidate(pattern)) ); } // 배치 무효화 (대량 캐시 무효화 시 성능 최적화) async batchInvalidate(invalidations: CacheInvalidation[]): Promise { const batchSize = 100; for (let i = 0; i < invalidations.length; i += batchSize) { const batch = invalidations.slice(i, i + batchSize); await Promise.all( batch.map(async (invalidation) => { try { await this.cacheManager.invalidate( invalidation.pattern, invalidation.options ); } catch (error) { console.error(`Cache invalidation failed:`, error); } }) ); // 배치 간 짧은 대기 (과부하 방지) if (i + batchSize < invalidations.length) { await this.delay(10); } } } private delay(ms: number): Promise { return new Promise(resolve => setTimeout(resolve, ms)); } }

// 임계값 확인 및 알림 if (hitRate < this.alertThresholds.minHitRate) { this.sendAlert({ type: 'low_hit_rate', layer, currentRate: hitRate, threshold: this.alertThresholds.minHitRate }); } } // 캐시 응답 시간 기록 recordResponseTime(layer: string, responseTime: number): void { this.metrics.recordResponseTime(layer, responseTime); } // 핫 키 감지 및 분산 제안 const hotKeys = this.detectHotKeys(stats); if (hotKeys.length > 0) { recommendations.push({ type: 'distribute_hot_keys', priority: 'medium', description: '핫 키를 여러 키로 분산하여 부하를 줄이세요.', affectedKeys: hotKeys }); } return recommendations; } // 캐시 워밍 (사전 로딩) async warmupCache(keys: string[]): Promise { const results = { successful: 0, failed: 0, errors: [] }; for (const key of keys) { try { // 데이터 소스에서 로드하여 캐시에 저장 const data = await this.loadFromDataSource(key); await this.cacheManager.set(key, data); results.successful++; } catch (error) { results.failed++; results.errors.push({ key, error: error.message }); } } return results; } // 캐시 상태 건강성 검사 async healthCheck(): Promise { const health = { overall: 'healthy' as HealthStatus, checks: [] as HealthCheck[] }; 'healthy' : 'warning', details: memoryStats }); // 전체 시스템 성능 확인 const hitRate = this.metrics.getOverallHitRate(); health.checks.push({ component: 'Cache Performance', status: hitRate > 0.7 ? 'healthy' : 'warning', hitRate }); return health; } }

// CDN 및 브라우저 캐시 관리 class CDNAndBrowserCacheManager { private cdnProvider: CDNProvider; private cacheHeaders: CacheHeaderGenerator; // 정적 자원 캐시 헤더 설정 setStaticAssetCacheHeaders(response: Response, assetType: string): void { const cacheConfig = this.getCacheConfigForAssetType(assetType); // 브라우저 캐시 설정 response.setHeader('Cache-Control', `public, max-age=${cacheConfig.maxAge}, s-maxage=${cacheConfig.sMaxAge}` ); // ETag 설정 (파일 내용 기반) const etag = this.generateETag(response.body); response.setHeader('ETag', etag); // Last-Modified 설정 response.setHeader('Last-Modified', new Date().toUTCString()); // CDN 캐시 키 변형 방지 response.setHeader('Vary', 'Accept-Encoding'); } private getCacheConfigForAssetType(assetType: string): CacheConfig { const configs = { 'css': { maxAge: 31536000, sMaxAge: 31536000 }, // 1년 'js': { maxAge: 31536000, sMaxAge: 31536000 }, // 1년 'images': { maxAge: 2592000, sMaxAge: 2592000 }, // 30일 'fonts': { maxAge: 31536000, sMaxAge: 31536000 }, // 1년 'html': { maxAge: 300, sMaxAge: 3600 }, // 5분/1시간 'api': { maxAge: 0, sMaxAge: 60 } // 노캐시/1분 }; return configs[assetType] || configs['html']; } // 조건부 요청 처리 (304 Not Modified) handleConditionalRequest( request: Request, response: Response, resourceInfo: ResourceInfo ): boolean { const ifNoneMatch = request.headers['if-none-match']; const ifModifiedSince = request.headers['if-modified-since']; // ETag 비교 if (ifNoneMatch && ifNoneMatch === resourceInfo.etag) { response.status(304).end(); return true; } // Last-Modified 비교 if (ifModifiedSince) { const modifiedSince = new Date(ifModifiedSince); const lastModified = new Date(resourceInfo.lastModified); if (lastModified <= modifiedSince) { response.status(304).end(); return true; } } return false; } // CDN 캐시 무효화 async invalidateCDNCache(urls: string[]): Promise { try { const invalidationId = await this.cdnProvider.createInvalidation(urls);
