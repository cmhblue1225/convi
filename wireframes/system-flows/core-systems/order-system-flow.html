// 주문 아이템 검증 const validateOrderItems = async (items: OrderItem[], storeId: string) => { const validationResults = await Promise.all( items.map(async (item) => { // 상품 존재 여부 확인 const product = await getProduct(item.productId); if (!product) { throw new OrderValidationError('PRODUCT_NOT_FOUND', item.productId); } // 매장 상품 확인 const storeProduct = await getStoreProduct(storeId, item.productId); if (!storeProduct || !storeProduct.isAvailable) { throw new OrderValidationError('PRODUCT_UNAVAILABLE', item.productId); } // 재고 확인 if (storeProduct.stockQuantity < item.quantity) { throw new OrderValidationError('INSUFFICIENT_STOCK', { productId: item.productId, requested: item.quantity, available: storeProduct.stockQuantity }); } return { item, storeProduct, valid: true }; }) ); return validationResults; };

// 재고 임시 예약 (Pessimistic Locking) const reserveStockForOrder = async (orderItems: OrderItem[]) => { const reservationId = generateReservationId(); await db.transaction(async (trx) => { for (const item of orderItems) { // 행 단위 잠금으로 동시성 제어 const storeProduct = await trx('store_products') .where({ store_id: item.storeId, product_id: item.productId }) .forUpdate() .first(); if (storeProduct.stock_quantity < item.quantity) { throw new InsufficientStockError(item); } // 예약 테이블에 기록 await trx('stock_reservations').insert({ id: generateUUID(), reservation_id: reservationId, store_product_id: storeProduct.id, quantity: item.quantity, expires_at: new Date(Date.now() + RESERVATION_TIMEOUT), created_at: new Date() }); } }); // 예약 만료 스케줄링 scheduleReservationExpiry(reservationId, RESERVATION_TIMEOUT); return reservationId; };

동시성 제어: FOR UPDATE 락 사용

주문 금액 계산

// 주문 총액 계산 엔진 class OrderCalculator { private order: Order; private discounts: Discount[] = []; calculateTotal(): OrderTotal { // 1. 기본 소계 계산 const subtotal = this.calculateSubtotal(); // 2. 할인 적용 (쿠폰, 포인트, 회원할인) const discountAmount = this.applyDiscounts(subtotal); // 3. 세금 계산 const taxAmount = this.calculateTax(subtotal - discountAmount); // 4. 배송비 계산 const deliveryFee = this.calculateDeliveryFee(); // 5. 최종 금액 const totalAmount = subtotal - discountAmount + taxAmount + deliveryFee; return { subtotal, discountAmount, taxAmount, deliveryFee, totalAmount }; } private applyDiscounts(subtotal: number): number { let totalDiscount = 0; // 쿠폰 할인 우선 적용 totalDiscount += this.applyCoupons(subtotal); // 포인트 할인 totalDiscount += this.applyPoints(subtotal - totalDiscount); // 회원 등급 할인 totalDiscount += this.applyMembershipDiscount(subtotal - totalDiscount); return Math.min(totalDiscount, subtotal * 0.8); // 최대 80% 할인 제한 } }

계산 순서: 소계 → 할인 → 세금 → 배송비

할인 제한: 최대 80% 할인 방지

정확성: 소수점 처리 및 반올림 규칙

-- 주문 상태 변경 트리거 CREATE OR REPLACE FUNCTION handle_order_status_change() RETURNS TRIGGER AS $$ DECLARE notification_data JSONB; BEGIN -- 상태 이력 기록 INSERT INTO order_status_history ( order_id, status, changed_at, changed_by, notes ) VALUES ( NEW.id, NEW.status, NOW(), NEW.updated_by, NEW.status_notes ); -- 완료 시 재고 차감 IF NEW.status = 'completed' AND OLD.status != 'completed' THEN UPDATE store_products SET stock_quantity = stock_quantity - oi.quantity, updated_at = NOW() FROM order_items oi WHERE oi.order_id = NEW.id AND store_products.id = oi.store_product_id; -- 매출 요약 업데이트 PERFORM update_daily_sales_summary(NEW.store_id, NEW.total_amount); END IF; -- 실시간 알림 생성 notification_data = jsonb_build_object( 'order_id', NEW.id, 'status', NEW.status, 'store_id', NEW.store_id, 'customer_id', NEW.customer_id ); PERFORM pg_notify('order_status_change', notification_data::text); RETURN NEW; END; $$ LANGUAGE plpgsql; CREATE TRIGGER order_status_change_trigger AFTER UPDATE ON orders FOR EACH ROW EXECUTE FUNCTION handle_order_status_change();

자동 실행: 상태 변경 시 즉시 트리거

데이터 무결성: 원자적 재고 차감

// 주문 처리 에러 핸들러 class OrderErrorHandler { static async handleOrderError(error: OrderError, context: OrderContext) { // 에러 로깅 logger.error({ error: error.message, code: error.code, orderId: context.orderId, customerId: context.customerId, storeId: context.storeId, stack: error.stack, timestamp: new Date() }); // 상황별 복구 로직 switch (error.code) { case 'INSUFFICIENT_STOCK': await this.handleStockShortage(context); break; case 'PAYMENT_FAILED': await this.handlePaymentFailure(context); break; case 'STORE_CLOSED': await this.handleStoreClosure(context); break; case 'DATABASE_ERROR': await this.handleDatabaseError(context); break; } // 고객에게 적절한 에러 메시지 전송 await this.notifyCustomerOfError(context, error); } private static async handleStockShortage(context: OrderContext) { // 재고 예약 해제 await releaseStockReservation(context.reservationId); // 대체 상품 제안 const alternatives = await findAlternativeProducts(context.unavailableItems); // 고객에게 대안 제시 await sendAlternativeProductSuggestion(context.customerId, alternatives); } }
