// 저재고 감지 알고리즘 class LowStockDetector { async checkStockLevels(storeId: string, productId: string) { const storeProduct = await this.getStoreProduct(storeId, productId); const currentStock = storeProduct.stock_quantity; const minimumStock = storeProduct.minimum_stock; const reservedStock = await this.getReservedStock(storeId, productId); const availableStock = currentStock - reservedStock; if (availableStock <= 0) { // 품절 상황 await this.handleOutOfStock(storeProduct); } else if (availableStock <= minimumStock) { // 저재고 상황 await this.handleLowStock(storeProduct, availableStock); } else if (availableStock > storeProduct.maximum_stock) { // 과재고 상황 await this.handleOverStock(storeProduct, availableStock); } } private async handleLowStock( storeProduct: StoreProduct, availableStock: number ) { // 알림 생성 await this.createNotification({ type: 'low_stock', storeId: storeProduct.store_id, productId: storeProduct.product_id, currentStock: availableStock, minimumStock: storeProduct.minimum_stock, urgency: this.calculateUrgency(availableStock, storeProduct.minimum_stock) }); // 자동 발주 확인 if (storeProduct.auto_order_enabled) { await this.triggerAutoOrder(storeProduct); } } }

// 스마트 발주량 계산 class AutoOrderCalculator { async calculateOptimalOrderQuantity(storeProduct: StoreProduct): Promise { // 판매 패턴 분석 const salesPattern = await this.analyzeSalesPattern( storeProduct.store_id, storeProduct.product_id, 30 // 지난 30일 데이터 ); // 기본 계산 요소 const dailyAverageConsumption = salesPattern.dailyAverage; const leadTimeDays = await this.getSupplierLeadTime(storeProduct.product_id); const safetyStockDays = this.calculateSafetyStockDays(salesPattern.variance); // 최적 발주량 = (일평균 소비량 × 리드타임) + 안전재고 - 현재재고 const optimalQuantity = Math.max(0, (dailyAverageConsumption * leadTimeDays) + (dailyAverageConsumption * safetyStockDays) - storeProduct.stock_quantity ); // 경제적 주문량(EOQ) 고려 const economicOrderQuantity = this.calculateEOQ( salesPattern.annualDemand, await this.getOrderingCost(storeProduct.product_id), await this.getHoldingCost(storeProduct.product_id) ); // 최종 발주량 결정 return Math.min( Math.max(optimalQuantity, economicOrderQuantity), storeProduct.maximum_stock - storeProduct.stock_quantity ); } private calculateEOQ(annualDemand: number, orderingCost: number, holdingCost: number): number { return Math.sqrt((2 * annualDemand * orderingCost) / holdingCost); } }

최적화: EOQ 모델 적용

// 재고 실사 프로세스 class StockCountManager { async initiateStockCount(storeId: string, countType: 'full' | 'cycle' | 'spot') { const stockCountId = await this.createStockCountRecord(storeId, countType); // 재고 동결 (실사 중 거래 차단) await this.freezeInventory(storeId, stockCountId); // 실사 대상 상품 목록 생성 const products = await this.getProductsForCount(storeId, countType); // 실사 태스크 생성 const tasks = products.map(product => ({ stockCountId, productId: product.id, systemQuantity: product.stock_quantity, countedQuantity: null, status: 'pending' })); await this.createStockCountTasks(tasks); return stockCountId; } async processStockCountResults(stockCountId: string, results: CountResult[]) { const discrepancies = []; for (const result of results) { const systemQty = result.systemQuantity; const countedQty = result.countedQuantity; const variance = countedQty - systemQty; if (Math.abs(variance) > 0) { discrepancies.push({ productId: result.productId, variance, adjustmentReason: 'stock_count_adjustment' }); // 시스템 재고를 실제 재고로 조정 await this.adjustInventory( result.storeId, result.productId, variance, 'stock_count_adjustment', stockCountId ); } } // 재고 동결 해제 await this.unfreezeInventory(stockCountId); // 실사 보고서 생성 await this.generateStockCountReport(stockCountId, discrepancies); } }

// 재고 분석 엔진 class InventoryAnalytics { async generateInventoryInsights(storeId: string, period: DateRange) { // 재고 회전율 분석 const turnoverAnalysis = await this.calculateInventoryTurnover(storeId, period); // 재고 가치 분석 const valueAnalysis = await this.calculateInventoryValue(storeId); // 데드스톡 분석 const deadStockAnalysis = await this.identifyDeadStock(storeId, 90); // 90일 // 계절성 분석 const seasonalityAnalysis = await this.analyzeSeasonality(storeId, period); // 재고 효율성 점수 const efficiencyScore = this.calculateEfficiencyScore({ turnoverRatio: turnoverAnalysis.averageTurnover, stockoutRate: await this.calculateStockoutRate(storeId, period), excessInventoryRate: await this.calculateExcessInventoryRate(storeId), accuracyRate: await this.calculateInventoryAccuracy(storeId, period) }); return { turnover: turnoverAnalysis, value: valueAnalysis, deadStock: deadStockAnalysis, seasonality: seasonalityAnalysis, efficiency: efficiencyScore, recommendations: await this.generateRecommendations(storeId, { turnoverAnalysis, deadStockAnalysis, efficiencyScore }) }; } private async generateRecommendations( storeId: string, analysis: AnalysisResult ): Promise { const recommendations = []; // 저회전 상품 처리 권장 if (analysis.turnoverAnalysis.lowTurnoverProducts.length > 0) { recommendations.push({ type: 'reduce_slow_movers', priority: 'medium', products: analysis.turnoverAnalysis.lowTurnoverProducts, action: '프로모션 또는 재고 감축 권장' }); } // 데드스톡 처리 권장 if (analysis.deadStockAnalysis.deadStockValue > 100000) { recommendations.push({ type: 'clear_dead_stock', priority: 'high', value: analysis.deadStockAnalysis.deadStockValue, action: '데드스톡 클리어런스 세일 권장' }); } return recommendations; } }

// 재고 예약 관리 class InventoryReservationManager { private readonly RESERVATION_TIMEOUT = 15 * 60 * 1000; // 15분 async reserveStock(reservationRequest: ReservationRequest): Promise { const reservationId = generateUUID(); const trx = await db.transaction(); try { // 현재 가용 재고 확인 (예약된 재고 제외) const availableStock = await this.getAvailableStock( reservationRequest.storeId, reservationRequest.productId, trx ); if (availableStock < reservationRequest.quantity) { throw new InsufficientStockError( `가용 재고: ${availableStock}, 요청량: ${reservationRequest.quantity}` ); } // 재고 예약 기록 await trx('stock_reservations').insert({ id: reservationId, store_id: reservationRequest.storeId, product_id: reservationRequest.productId, quantity: reservationRequest.quantity, expires_at: new Date(Date.now() + this.RESERVATION_TIMEOUT), reference_type: reservationRequest.referenceType, reference_id: reservationRequest.referenceId, created_by: reservationRequest.createdBy }); await trx.commit(); // 예약 만료 스케줄링 this.scheduleReservationExpiry(reservationId, this.RESERVATION_TIMEOUT); return reservationId; } catch (error) { await trx.rollback(); throw error; } } async confirmReservation(reservationId: string): Promise { const reservation = await this.getReservation(reservationId); if (!reservation || reservation.status !== 'active') { throw new InvalidReservationError('예약을 찾을 수 없거나 만료됨'); } // 실제 재고 차감 await this.finalizeStockDeduction(reservation); // 예약 상태 변경 await this.updateReservationStatus(reservationId, 'confirmed'); } }
